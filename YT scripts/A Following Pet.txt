--Delta, PushPositionOutWalls is just like IsBlockLocked but use position and scan a square with sides of sqrt(x^2+y^2+z^2) for biggest coverage/centering between narrow path
local Part = script:WaitForChild('Part')
function IsBlockLocked(filter)
 local scanLeft, scanRight, ScanBack, ScanFront, ScanDown, ScanUp = FilterResult(workspace:Blockcast(Part.CFrame + Part.CFrame.RightVector*Part.Size.X/2, Part.Size,-Part.CFrame.RightVector*Part.Size.X, filter)) , FilterResult(workspace:Blockcast(Part.CFrame - Part.CFrame.RightVector*Part.Size.X/2, Part.Size,Part.CFrame.RightVector*Part.Size.X, filter)) , FilterResult(workspace:Blockcast(Part.CFrame + Part.CFrame.LookVector*Part.Size.Z/2, Part.Size,-Part.CFrame.LookVector*Part.Size.Z, filter)) , FilterResult(workspace:Blockcast(Part.CFrame - Part.CFrame.LookVector*Part.Size.Z/2, Part.Size,Part.CFrame.LookVector*Part.Size.Z, filter)) , FilterResult(workspace:Blockcast(Part.CFrame + Part.CFrame.UpVector*Part.Size.Y/2, Part.Size,-Part.CFrame.UpVector*Part.Size.Y, filter)) , FilterResult(workspace:Blockcast(Part.CFrame - Part.CFrame.UpVector*Part.Size.Y/2, Part.Size,Part.CFrame.UpVector*Part.Size.Y, filter))
 if typeof(scanLeft or scanRight or ScanBack or ScanFront or ScanDown or ScanUp) ~= 'RaycastResult' then return false end
 local clipout = Vector3.new(-(Part.Size.X - (scanLeft or {Distance = Part.Size.X}).Distance) + (Part.Size.X - (scanRight or {Distance = Part.Size.X}).Distance),- (Part.Size.Y - (ScanDown or {Distance = Part.Size.Y}).Distance) + (Part.Size.Y - (ScanUp or {Distance = Part.Size.Y}).Distance), -(Part.Size.Z - (ScanBack or {Distance = Part.Size.Z}).Distance) + (Part.Size.Z - (ScanFront or {Distance = Part.Size.Z}).Distance) )
 Part.CFrame = Part.CFrame + Part.CFrame.Rotation:PointToWorldSpace(clipout)
 return true
end
