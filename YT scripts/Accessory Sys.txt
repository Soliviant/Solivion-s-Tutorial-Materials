Code in the accessory_system:
local ReplicatedStorage = game:GetService('ReplicatedStorage')
local ACCESSORIES = ReplicatedStorage:WaitForChild('ACCESSORIES')

function ffcac(par, n, c)
 if typeof(par) ~= 'Instance' or typeof(n) ~= 'string' or typeof(c) ~= 'string' then return end
 for _, v in pairs(par:GetChildren()) do
  if v.Name:match(n) and (v.ClassName == c or v:IsA(c)) then
   return v
  end
 end
 return false
end

local function createAccessory(chr, accessoryModule, attachmentPartName, namae)
 if typeof(accessoryModule) ~= 'Instance' or accessoryModule.ClassName ~= 'ModuleScript' or typeof(attachmentPartName) ~= 'string' then return end
 local attachmentPart = ffcac(chr, attachmentPartName, 'BasePart')
 if not attachmentPart then return true end
 
 local ran = pcall(function()
  local accessoryData = require(accessoryModule)
  local accessory = Instance.new("Accessory")
  accessory.Name = namae or accessoryData.Name or accessoryModule.Name or 'NO_NAME!'
  accessory.Parent = chr

  for i, v in accessoryData do
   local part = Instance.new('Part', accessory)
   part.Name = v.Name or 'parte'
   part.Color = v.Color or Color3.new(1,1,1)
   part.Material = v.Material or Enum.Material.SmoothPlastic
   part.Shape = v.Shape or Enum.PartType.Block 
   part.Size = v.Size or Vector3.new(1,1,1) -- I'll add extra adjustments here to handle missing datas
   part.Transparency = v.Transparency or 0
   part.CanCollide = v.CanCollide or false
   part.CFrame = attachmentPart.CFrame * (v.WeldC0 or CFrame.new())
   part.Anchored = v.Anchored or false
   local w = Instance.new('Weld', part)
   w.Part0 = attachmentPart
   w.Part1 = part
   w.C0 = v.WeldC0 or CFrame.new()
  end
 end)
 
 return ran or false
end


workspace.ChildAdded:Connect(function(chld)
 if chld:IsA('Model') and chld:FindFirstChildOfClass('Humanoid') then
  pcall(function() createAccessory(chld, game.ReplicatedStorage.ACCESSORIES.Head.Headphone, 'Head', 'Th') end)
 end
end)
