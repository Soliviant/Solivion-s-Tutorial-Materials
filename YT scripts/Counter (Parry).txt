local chr

function filter(v)
 if typeof(v) ~= 'Instance' then return end
 if v:IsA("BasePart") or v:IsA('Decal') or v:IsA("Texture") then
  game:GetService('TweenService'):Create(v, TweenInfo.new(1), {Transparency = 1}):Play()
  if v:IsA("BasePart") or v.ClassName:match('Part') then
   v.CanCollide = false 
   v.CanTouch = false
   v.CanQuery = false
   v.Anchored = true
   v.CastShadow = false
   v.Massless = true
   v.CollisionGroup = "__NOCOLLIDE"
   v.AssemblyLinearVelocity = Vector3.zero
   v.AssemblyAngularVelocity = Vector3.zero
  end
 elseif v:IsA('Highlight') or v:IsA('ParticleEmitter') or v:IsA('Sound') or v:IsA('Humanoid') then -- Humanoid FORCES the Character to be collidable
  v:Destroy() end
end
function Aft(chr)
 if typeof(chr) ~= 'Instance' then return end
 chr.Archivable = true
 local clone  = chr:Clone()
 task.spawn(function() for i,v in clone:GetDescendants() do
  filter(v)
 end end)
 clone.Parent = workspace
 game:GetService('Debris'):AddItem(clone, 1)
 task.spawn(function()
  for i,v in clone:GetDescendants() do
   filter(v)
  end
 end
 )
end
local lasthum
