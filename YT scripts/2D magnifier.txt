-- Update magnifier position and image offset
function update() -- I'm sure u get what real() is at this point, if not, real checks if typeof the variable is an instance alongside its parent
 --mousePos obviously mouse Position, magSize is scalar way of telling of 1x 2x 3x zoom typa stuff,magnifierImage is the clone, mainImage is the main image(obviously, please put it at the root and scale it as you wish), magnifierRadius is basically how big the (magnifierFrame) is in pixel
 if not real(mainImageLabel) or not real(magnifierFrame) or not real(magnifiedImage) then return end -- MAKE SURE THE BASEIMAGE AND THE CLONED(FOR MAGNIFIED LOOK) USES SAME SCALETYPE
 local inset = guiService:GetGuiInset() -- sometimes it's not fullscreen(no inset), we gotta account that for the mouse(since this mf is calculated as full screen(no inset))
 local adjustedMouseX,adjustedMouseY = mousePos.X - inset.X,mousePos.Y - inset.Y
    -- center the magnifier frame at mouse
    magnifierFrame.Position = UDim2.fromOffset(adjustedMouseX - magnifierRadius, adjustedMouseY - magnifierRadius) 
 magnifiedImage.Size = UDim2.fromOffset(mainImageLabel.AbsoluteSize.X * Magsize, mainImageLabel.AbsoluteSize.Y * Magsize)
 --to keep it a gurt, we gotta counter the frame displacement relative to the global/main Image, so it can "Stay in the same place"
 magnifiedImage.Position = UDim2.fromOffset(-(adjustedMouseX - mainImageLabel.AbsolutePosition.X) * Magsize + magnifierRadius, -(adjustedMouseY -  mainImageLabel.AbsolutePosition.Y) * Magsize + magnifierRadius)
end
