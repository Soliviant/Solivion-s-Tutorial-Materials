function applySkew(cframe, skewX, skewY, skewZ) -- SkewZ is barely visible  btw, it got cancelled out during the calculation and my brain wouldn't want to try for the love of my mental health.
 if skewX == 0 and skewY == 0 and skewZ == 0 then
  return cframe
 end
 
 local pos = cframe.Position
 local look = cframe.LookVector
 local up = cframe.UpVector
 local right = cframe.RightVector
 
 -- Apply skew only to right and up vectors relative to each other
 -- This creates the visual distortion while preserving the look direction
 -- skewX: skews up vector toward right (horizontal shear)
 -- skewY: skews right vector toward up (vertical shear)
 -- skewZ: skews both toward look direction (depth/perspective shear)
 local skewedRight = right + (up * skewY) + (look * skewZ)
 local skewedUp = up + (right * skewX) + (look * skewZ)
 
 -- Compute the new look vector from cross product
 -- For proper skew, we want to preserve the original look direction
 -- So we'll use the original look vector and adjust right/up to be perpendicular to it
 local newLook = look
 
 -- Make skewedRight and skewedUp perpendicular to the original look
 -- This preserves the camera's facing direction while still creating skew
 skewedRight = skewedRight - (skewedRight:Dot(look)) * look
 skewedUp = skewedUp - (skewedUp:Dot(look)) * look
 
 -- Reconstruct CFrame with corrected axes
 return CFrame.fromMatrix(pos, skewedRight, skewedUp)
end
