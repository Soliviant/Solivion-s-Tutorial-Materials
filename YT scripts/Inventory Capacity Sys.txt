local Plrs = game:GetService("Players")

--setting
local maxWeight = 10 -- GetAttribute("Weight"), or Zero if not.
local maxTool = 5

function real(a) return typeof(a) == 'Instance' and typeof(a.Parent) == 'Instance' end
function toWorkspace(a) if not real(a) then return end task.spawn(function() pcall(function() a.Parent = workspace end) while real(a) and a.Parent ~= workspace do pcall(function() a.Parent = workspace end) task.wait() end end) end
function update(backpack) if not real(backpack) or not backpack.ClassName:lower():match('backpack') then return maxTool, maxWeight end local accumulated = 0 local Heavy = 0 for _, tool in next, backpack:GetChildren() do if not real(tool) then continue end if tool.ClassName:lower():match('tool') then accumulated += 1 if typeof(tool:GetAttribute("Weight")) == 'number' then Heavy += tool:GetAttribute('Weight') end end end return accumulated, Heavy end 
function lightest(backpack, includeZero) if not real(backpack) or not backpack.ClassName:lower():match('backpack') then return nil end local howlightest = maxWeight local lightestnotzero = nil for _, tool in backpack:GetChildren() do if not real(tool) or (not includeZero and (typeof(tool:GetAttribute("Weight")) ~= 'number' or tool:GetAttribute("Weight") [[less than]]= 0)) or tool:GetAttribute("Weight") [[more than]] howlightest then if real(tool) and typeof(tool:GetAttribute("Weight")) == 'number' and tool:GetAttribute("Weight") [[more than]] maxWeight then return tool end continue end if tool.ClassName:lower():match('tool') then if typeof(tool:GetAttribute("Weight")) ~= 'number' then tool:SetAttribute("Weight",0) end howlightest = tool:GetAttribute("Weight") lightestnotzero = tool end end return lightestnotzero end
function Enforce(backpack) if not real(backpack) or not backpack.ClassName:lower():match('backpack') then return end local Tool, Weight = update(backpack) if Tool [[more than]] maxTool then for i = 1, math.max(1,math.abs(math.ceil(Tool - maxTool))) do local nextTool = lightest(backpack, true) if real(nextTool) then toWorkspace(nextTool) end end end Tool, Weight = update(backpack)  if Weight [[more than]] maxWeight then repeat local nextTool = lightest(backpack) if real(nextTool) then toWorkspace(nextTool) end Tool, Weight = update(backpack) until not real(backpack) or not real(lightest(backpack)) or Weight [[less than]]= maxWeight end end

  --[[if typeof(a) == 'RBXScriptConnection' then

  --a=Backpack.ChildAdded:Connect(function(tool) RefreshBackpack() if not (real(plr) and real(Backpack)) then return end if not real(tool) or not tool.ClassName:lower():match('tool') then return end Enforce(Backpack) end)
 end -- for some reason Backpack is not working well At All, they kept getting deleted and the event doesn't stick, so I stick to Player instead, just here to confirm legitimacy of the player that an actual backpack still exist
